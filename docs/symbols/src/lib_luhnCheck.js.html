1 var assertString = require("../lib/isString");
  2 
  3 /**
  4  * @exports luhnCheck
  5  * @desc Checks if the card number is valid or not.
  6  * @param {String} cardNum
  7  * @returns {Boolean}
  8  */
  9 exports.luhnCheck = (cardNum)=>{
 10     assertString.isString(cardNum);
 11     var numericDashRegex = /^[\d\-\s]+$/
 12     if (!numericDashRegex.test(cardNum)) return false;
 13 
 14     var nCheck = 0, nDigit = 0, bEven = false;
 15     var strippedField = cardNum.replace(/\D/g, "");
 16 
 17     for (var n = strippedField.length - 1; n >= 0; n--) {
 18         var cDigit = strippedField.charAt(n);
 19         nDigit = parseInt(cDigit, 10);
 20         if (bEven) {
 21             if ((nDigit *= 2) > 9) nDigit -= 9;
 22         }
 23 
 24         nCheck += nDigit;
 25         bEven = !bEven;
 26     }
 27 
 28   	return (nCheck % 10) === 0;
 29 }
