1 /**
  2  * @exports isDate
  3  * @desc This function only recognize ISO format defined by ECMA Script ECMA-256
  4  * https://www.ecma-international.org/ecma-262/6.0/#sec-date.parse
  5  * @param {String} dateValue
  6  * @returns {Boolean}
  7  */
  8 exports.isDate = dateValue=> {
  9     let validateDate = Date.parse(new Date(dateValue));
 10     if(isNaN(validateDate)){
 11         return false;
 12     }else{
 13         return true;
 14     }
 15 }
