  3 const assert = require("assert");
  4 var isDate = require("../fns/isDate");
  5 
  6 describe("Unit test for isDate function", ()=>{
  7 
  8     it("The function should return true on valid dates", ()=>{
  9         let validDate = isDate.isDate("10 06 2014");
 10         assert.strictEqual(validDate, true);
 11     });
 12 
 13     it("The function should return true on complex dates", ()=>{
 14         let validDate2 = isDate.isDate("2011-10-10T14:48:00.000+09:00");
 15         let validDate3 = isDate.isDate("2011-10-10");
 16         let validDate4 = isDate.isDate("2011-10-10T14:48:00");
 17         let validDate5 = isDate.isDate("Thu, 01 Jan 1970 00:00:00 GMT-0400")
 18         let validDate6 = isDate.isDate("Aug 9, 1995")
 19         assert.strictEqual(validDate2, true);
 20         assert.strictEqual(validDate3, true);
 21         assert.strictEqual(validDate4, true);
 22         assert.strictEqual(validDate5, true);
 23         assert.strictEqual(validDate6, true);
 24     });
 25 
 26     it("The function should return false on invlaid dates", ()=>{
 27         let validDate = isDate.isDate("Tell them, winter came for House Frey");
 28         assert.strictEqual(validDate, false);
 29     });
 30 })
