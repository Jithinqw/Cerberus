1 var assertString = require("../lib/isString");
  2 const postalPattern = {
  3     IL: /^\d{5}$/,
  4     IN: /^\d{6}$/,
  5     US: /^\d{5}(-\d{4})?$/
  6 }
  7 
  8 /**
  9  * @exports isPostalCodeValid
 10  * @param {String} postalCode
 11  * @param {String} countryCode
 12  * @returns {Boolean}
 13  */
 14 exports.isPostalCodeValid = (postalCode, countryCode)=>{
 15     assertString.isString(postalCode);
 16     if (countryCode.toUpperCase() in postalPattern === true){
 17         return (postalPattern[countryCode].test(postalCode));
 18     }else{
 19         throw new Error (`Invalid country code ${countryCode}`);
 20     }
 21 }
