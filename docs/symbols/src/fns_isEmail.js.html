  5 /**
  6  * @function isUsernameValid
  7  * @param {*} email 
  8  * @returns {Boolean}
  9  */
 10 var isUsernameValid = email=> {
 11     assertString.isString(email);
 12     let username = splitUsername.getUsername(email);
 13     if(/[\.";<>]/.test(username.charAt(0)) === false){
 14         return true;
 15     }else{
 16         return false;
 17     }
 18 }

 30 /**
 31  * @function restrictedDomainCheck
 32  * @param {*} email 
 33  * @returns {Boolean}
 34  */
 35 var restrictedDomainCheck = email=> {
 36     assertString.isString(email);
 37     var result = true;
 38     let domain = email.split("@");
 39     for(let i=0; i <= restrictedDomains["domains"].length; i++){
 40         if((restrictedDomains["domains"][i]) === domain[1]){
 41             result = true;
 42         }else{
 43             result = false;
 44         }
 45     }
 46     return result;
 47 }

 49 /**
 50  * @exports isEmailValid
 51  * @desc Checks if the email is valid accoriding to the default or options config.
 52  * Checking is done using spec defined in RFC 5321.
 53  * @param {String} email
 54  * @return {Boolean}
 55  */
 56 exports.isEmailValid = email=> {
 57     assertString.isString(email);
 58     if(emailChecker(email) === true){
 59         if(restrictedDomainCheck(email) === false){
 60             if(isUsernameValid(email) === true){
 61                 return true
 62             }
 63         }
 64     }else{
 65         return false;
