1 var normEmail = require("../lib/emailDomains.json");
  2 var emailValidation = require("./isEmail");
  3 var assertString = require("../lib/isString");
  4 
  5 /**
  6  * @exports normalizeEmail
  7  * @desc Checks if the email is belonging to a particular domain or not.
  8  * @param {String} email 
  9  * @param {String} domainName
 10  */
 11 exports.normalizeEmail = (email, domainName)=>{
 12     assertString.isString(email);
 13     if(emailValidation.emailValidator(email) === false){
 14         return false
 15     }else{
 16         var nameMatch = email.split("@");
 17         for(let i=0; i<=normEmail[domainName].length; i++){
 18         if(nameMatch[1] == normEmail[domainName][i]){
 19             return true;
 20         }else{
 21             return false;
 22         }
 23         }
 24     }
 25 }
