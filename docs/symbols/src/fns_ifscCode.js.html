1 var assertString = require("../lib/isString");
  2 var IFSCCode = require("../lib/bank.json");

  4 /**
  5  * @exports isIFSCode
  6  * @param {String} code
  7  * @returns {Boolean}
  8  */
  9 var IFSCodeValidator = exports.isIFSCode = code=> {
 10     assertString.isString(code);
 11     const IFSC = /^(?=.*\d)(?=.*[0-9A-Z]).{11}$/;
 12     return IFSC.test(code);
 13 }
 14 
 15 /**
 16  * @exports getBankCode
 17  * @param { String } code 
 18  * @returns {String}
 19  */
 20 exports.getBankCode = (code)=> {
 21     assertString.isString(code);
 22     if(IFSCodeValidator(code) === true){
 23         return (code.substring(0,4))
 24     }else{
 25         return false;
 26     }
 27 }
