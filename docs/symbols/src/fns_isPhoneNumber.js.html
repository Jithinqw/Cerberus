 84 /**
 85  * @exports isPhoneNumber
 86  * @param {String} phoneNumber
 87  * @param {String} countryCode
 88  * @returns {Boolean}
 89  */
 90 exports.isPhoneNumber = (phoneNumber, countryCode)=> {
 91   assertString.isString(phoneNumber);
 92   if(countryCode in phones === true){
 93     return phones[countryCode].test(phoneNumber);
 94   }
 95 }
