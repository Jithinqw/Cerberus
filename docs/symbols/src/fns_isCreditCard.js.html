 14 /**
 15  * @exports detectCardType
 16  * @desc detects card type
 17  * @param { String } cardNumber
 18  * @returns { String }
 19  */
 20 exports.detectCardType = cardNumber =>{
 21     assertString.isString(cardNumber);
 22         var re = {
 23             electron: /^(4026|417500|4405|4508|4844|4913|4917)\d+$/,
 24             maestro: /^(5018|5020|5038|5612|5893|6304|6759|6761|6762|6763|0604|6390)\d+$/,
 25             dankort: /^(5019)\d+$/,
 26             interpayment: /^(636)\d+$/,
 27             unionpay: /^(62|88)\d+$/,
 28             visa: /^4[0-9]{12}(?:[0-9]{3})?$/,
 29             mastercard: /^5[1-5][0-9]{14}$/,
 30             amex: /^3[47][0-9]{13}$/,
 31             diners: /^3(?:0[0-5]|[68][0-9])[0-9]{11}$/,
 32             discover: /^6(?:011|5[0-9]{2})[0-9]{12}$/,
 33             jcb: /^(?:2131|1800|35\d{3})\d{11}$/
 34         }
 35     
 36         for(var key in re) {
 37             if(re[key].test(cardNumber)) {
 38                 return key
 39             }
 40         }
 41 }
